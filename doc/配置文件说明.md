# CCCC

## 配置文件

配置文件的参数如下：

```ini
[train]  #训练过程设置
#载入文件
load_file = save.txt
#是否载入文件
load_net = 0
#网络的batch，实际上要以脚本中的batch为准
Batch = 100
#测试的间隔次数
test_epoch = 1
#是否测试测试集（可能经过变换）
test_test = 1
#是否测试训练集（可能经过变换）
test_train = 0
#是否测试原始训练集
test_train_origin=1
#是否测试原始测试集
test_test_origin=0
#总训练的epoch
train_epochs = 10
#基础学习率
learn_rate_base =1e-4
#权重衰减
weight_decay = 5e-4
#动量因子，仅在sgd和nag求解器下有效
momentum = 0.9
#学习率调整方法，有fixed，steps，scale_inter，steps_auto等
lr_adjust_method = fixed
#学习率调整步骤，仅在steps方法下有效
lr_steps = 3
#每次下降的比例，仅在steps方法下有效
lr_step_decay=0.1
#调整方法为scale_inter的设定，以下设定表示在每个特殊的epoch的时候，学习率的值，中间的值由等比插值计算，两边的值则等于第一个和最后一个设定值
lr_inter_set=1 1, 2 10000, 5 10000, 6 0.1
#是否使用gpu
gpu = 0
#求解器，有sgd，adam，nag等
solver=adam
#是否输出log
output_log=1
#是否输出网络结果
output_net=1
#并行的设备个数
mp=1
#并行的设备顺序
mp_device=0
#保存的间隔次数，须为测试间隔次数的整数倍
save_epoch=5
#保存模型的签名字串
save_sign=MNIST
#保存文件名的格式，可用的选项有{epoch}，{date}，{time}等
save_format=save/save-{epoch}-{date}.txt
#若空闲显存比例低于此值则不能训练
need_free_mem=0.2
#数据类型，可选float和half，通常训练时只能使用float
data_type=float
#训练时间限制，单位为秒，注意不包含载入图片和首次测试耗费的时间
#若开启此功能，学习率调整策略建议只使用steps，否则现象可能会比较奇怪
time_limited=180
#是否对每个权重单独调用求解器更新（不合并为 all_weights_ 的连续缓冲）
#0（默认）：初始化完成后将所有权重合并到一块连续 GPU 内存，Solver 一次批量更新全部权重，效率更高
#1：每个权重保留独立缓冲，使用各自的 Solver 实例单独更新；适用于需要对不同权重指定不同求解器类型或学习率的场景
#注意：推理模式（train_epochs=0）下 Solver 不初始化，权重始终不合并，all_weights_ 保持为空
separate_update_weight=0

[data_preparer]  #训练集的设置，其设置需要参考训练集插件的内容
#插件的dll文件（发布用），以下三项均为特殊需求的情况
library_dll=..\x64\release\cccc-mnist.dll
#插件的dll文件（调试用）
library_dlld=cccc-mnist.dll
#插件的so文件（linux用）
library_so=libcccc-mnist.so
#插件的名称，如果不使用源码，可以仅设置此项
library=cccc-mnist
#是否变换
trans = 0
#以下为mnist插件中的设置
#数据集类型，1表示训练集，2表示测试集
type=1
#是否打乱
random_diff=0
#是否加入噪声
d_noise=0
#是否去除数字5和9，如需使用翻转和转置变换，5和9会与2和6混淆，不宜用于测试
remove59=0
#是否随机翻转
flip=0
#是否随机转置
transpose=0
#对比度变换范围
d_contrast=-0.20,0.2
#亮度变换范围
d_brightness=-0.2,0.2
#是否随机反色
reverse_color=0


[data_preparer2]  #测试集的设置，未设置的选项则使用data_preparer的设置
library_dll=cccc-mnist.dll
library_dlld=cccc-mnist.dll
library_so=libcccc-mnist.so
library=cccc-mnist
trans = 1
type=2
transpose = 0
remove59=0
d_contrast=-0.9,-0.8
d_brightness=-0.1,0.1
reverse_color=0

[net]
#网络结构脚本，可直接在此段用 structure= 指定，或用 structure0= / structure1= 指定多组网络
#旧版本的 [cifa] 段写法仍向后兼容

[cifa]
#旧格式：网络结构脚本写在此段的 structure= 键下，现推荐改写到 [net] structure=

```

## 准备器

读取数据集需要用户自己编写C++代码实现，这部分会在主线程中运行。

用户需要编写从磁盘中读取数据到Matrix，以及一些自定义的变换等。基类中已经包含了一些简单变换，如亮度、对比度、翻转转置等。

如需要自己编写，需参考cccc-mnist的代码。

若设置mnist=1，则表示使用mnist的训练数据（60000个）。若为2，则表示使用测试数据（10000个）。
